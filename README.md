<script>
// ── CONSTANTS ──
var CARS_KEY = 'elite_cars_v1';
var PIN_KEY  = 'elite_pin_v1';

// ── HELPERS ──
function getPin() { return localStorage.getItem(PIN_KEY) || '1234'; }

function loadCars() {
  try { return JSON.parse(localStorage.getItem(CARS_KEY)) || []; }
  catch(e) { return []; }
}

function saveCars(arr) {
  localStorage.setItem(CARS_KEY, JSON.stringify(arr));
}

function el(id) { return document.getElementById(id); }
function val(id) { return el(id).value.trim(); }

// ── STATE ──
var activeFilter = 'all';
var tapCount = 0;
var tapTimer = null;

// ── SECRET 5-TAP ──
el('brand-trigger').addEventListener('click', function() {
  tapCount++;
  clearTimeout(tapTimer);
  tapTimer = setTimeout(function() { tapCount = 0; }, 1800);
  if (tapCount >= 5) { tapCount = 0; openLogin(); }
});

// ── LOGIN ──
function openLogin() {
  clearPins();
  el('login-error').textContent = '';
  el('login-overlay').classList.add('open');
  setTimeout(function() { el('p1').focus(); }, 150);
}
function closeLogin() { el('login-overlay').classList.remove('open'); }
function clearPins() { ['p1','p2','p3','p4'].forEach(function(id){ el(id).value = ''; }); }

// PIN auto-advance
['p1','p2','p3','p4'].forEach(function(id, i, arr) {
  var input = el(id);
  input.addEventListener('input', function() {
    this.value = this.value.replace(/\D/g, '');
    if (this.value && i < arr.length - 1) el(arr[i+1]).focus();
    if (i === 3 && this.value) setTimeout(checkPin, 80);
  });
  input.addEventListener('keydown', function(e) {
    if (e.key === 'Backspace' && !this.value && i > 0) el(arr[i-1]).focus();
    if (e.key === 'Enter') checkPin();
  });
});

function checkPin() {
  var entered = ['p1','p2','p3','p4'].map(function(id){ return el(id).value; }).join('');
  if (entered === getPin()) {
    closeLogin();
    openAdmin();
  } else {
    el('login-error').textContent = 'Incorrect PIN. Try again.';
    clearPins();
    el('p1').focus();
  }
}

// ── ADMIN PANEL ──
function openAdmin() {
  renderAdminList();
  el('admin-overlay').classList.add('open');
}
function closeAdmin() {
  el('admin-overlay').classList.remove('open');
  renderGrid();
}

function switchTab(name, btn) {
  ['add','manage','settings'].forEach(function(t) {
    el('tab-' + t).style.display = (t === name) ? '' : 'none';
  });
  document.querySelectorAll('.admin-tab').forEach(function(b) { b.classList.remove('active'); });
  btn.classList.add('active');
  if (name === 'manage') renderAdminList();
}

// ── IMAGE PREVIEW ──
function previewImg(inputId, previewId) {
  var url = val(inputId);
  var img = el(previewId);
  if (url) {
    img.src = url;
    img.classList.add('visible');
    img.onerror = function() { img.classList.remove('visible'); };
  } else {
    img.classList.remove('visible');
    img.src = '';
  }
}

// ── AUTO STOCK NUMBER ──
function nextStock() {
  var cars = loadCars();
  if (!cars.length) return 'STK-001';
  var nums = cars.map(function(c) {
    var m = c.stock.match(/(\d+)$/);
    return m ? parseInt(m[1], 10) : 0;
  });
  return 'STK-' + String(Math.max.apply(null, nums) + 1).padStart(3, '0');
}

// ── SAVE NEW CAR ──
function saveCar() {
  var name  = val('f-name');
  var price = val('f-price');
  if (!name)  { showToast('⚠️ Car name is required');  return; }
  if (!price) { showToast('⚠️ Price is required'); return; }
  if (isNaN(Number(price)) || Number(price) < 0) { showToast('⚠️ Enter a valid price'); return; }

  var car = {
    id:     Date.now(),
    name:   name,
    price:  Number(price),
    stock:  val('f-stock') || nextStock(),
    type:   el('f-type').value,
    status: el('f-status').value,
    mileage: val('f-mileage'),
    gear:   el('f-gear').value,
    engine: val('f-engine'),
    fuel:   el('f-fuel').value,
    colour: val('f-colour'),
    drive:  val('f-drive'),
    photo:  val('f-photo'),
    checks: val('f-checks')
  };

  var cars = loadCars();
  cars.unshift(car);
  saveCars(cars);
  clearForm();
  renderAdminList();
  renderGrid();
  showToast('✅ Car added successfully!');
  switchTab('manage', el('tab-btn-manage'));
}

  ['f-name','f-price','f-stock','f-mileage','f-engine','f-colour','f-drive','f-photo','f-checks']

  el('f-preview').classList.remove('visible');
  el('f-preview').src = '';
  el('f-type').value   = 'sedan';
  el('f-status').value = 'available';
  el('f-gear').value   = 'Automatic';
  el('f-fuel').value   = 'Petrol';
}

// ── EDIT CAR ──
function editCar(id) {
  var cars = loadCars();
  var car = null;
  for (var i = 0; i < cars.length; i++) { if (cars[i].id === id) { car = cars[i]; break; } }
  if (!car) return;

  el('e-id').value     = id;
  el('e-name').value   = car.name;
  el('e-price').value  = car.price;
  el('e-stock').value  = car.stock;
  el('e-type').value   = car.type;
  el('e-status').value = car.status;
  el('e-mileage').value = car.mileage || '';
  el('e-gear').value   = car.gear    || 'Automatic';
  el('e-engine').value = car.engine  || '';
  el('e-fuel').value   = car.fuel    || 'Petrol';
  el('e-colour').value = car.colour  || '';
  el('e-drive').value  = car.drive   || '';
  el('e-photo').value  = car.photo   || '';
  el('e-checks').value = car.checks  || '';

  var ep = el('e-preview');
  if (car.photo) { ep.src = car.photo; ep.classList.add('visible'); }
  else { ep.classList.remove('visible'); ep.src = ''; }

  el('edit-overlay').classList.add('open');

function closeEdit() { el('edit-overlay').classList.remove('open'); }

function updateCar() {
  var id    = Number(el('e-id').value);
  var name  = val('e-name');
  var price = val('e-price');
  if (!name)  { showToast('⚠️ Name is required');  return; }
  if (!price) { showToast('⚠️ Price is required'); return; }

  var cars = loadCars();
  var idx = -1;
  for (var i = 0; i < cars.length; i++) { if (cars[i].id === id) { idx = i; break; } }
  if (idx === -1) { showToast('⚠️ Car not found'); return; }

  cars[idx] = {
    id:     id,
    name:   name,
    price:  Number(price),
    stock:  val('e-stock') || cars[idx].stock,
    type:   el('e-type').value,
    status: el('e-status').value,
    mileage: val('e-mileage'),
    gear:   el('e-gear').value,
    engine: val('e-engine'),
    fuel:   el('e-fuel').value,
    colour: val('e-colour'),
    drive:  val('e-drive'),
    photo:  val('e-photo'),
    checks: val('e-checks')
  };

  saveCars(cars);
  closeEdit();
  renderAdminList();
  renderGrid();
  showToast('✅ Car updated!');
}

// ── RENDER GRID ──
function renderGrid() {
  var cars     = loadCars();
  var grid     = el('car-grid');
  var noRes    = el('no-results');

  // Remove existing cards but keep the no-results div
  var existing = grid.querySelectorAll('.car-card');
  existing.forEach(function(c) { c.remove(); });

  var filtered = (activeFilter === 'all')
    ? cars
    : cars.filter(function(c) { return c.type === activeFilter; });

  noRes.style.display = (filtered.length === 0) ? 'block' : 'none';

  // Count only available cars for stock pill
  var availCount = cars.filter(function(c) { return c.status === 'available'; }).length;
  el('stock-num').textContent = availCount;

  filtered.forEach(function(car) {
    grid.insertBefore(buildCard(car), noRes);
  });
}

// ── BUILD A CAR CARD (no inline event strings — uses data-id) ──
function buildCard(car) {
  var typeLabels   = {sedan:'Sedan', suv:'SUV / 4x4', hatchback:'Hatchback', minivan:'Minivan', pickup:'Pickup'};
  var statusLabels = {available:'Available', reserved:'Reserved', sold:'Sold'};

  var typeLabel   = typeLabels[car.type]   || car.type;
  var statusLabel = statusLabels[car.status] || car.status;

  var checksRaw = car.checks || 'Engine OK, Gearbox OK, Brakes OK, AC Working';
  var checks = checksRaw.split(',').map(function(s) { return s.trim(); }).filter(Boolean)
    .map(function(s) { return '<div class="check-pill">' + escHTML(s) + '</div>'; }).join('');

  var waMsg = encodeURIComponent(
    'Hi, I am interested in the ' + car.name + ' (' + car.stock + ') at K' +
    Number(car.price).toLocaleString() + '. Is it still available?'
  );

  var card = document.createElement('div');
  card.className = 'car-card';
  card.dataset.type = car.type;
  card.dataset.id   = String(car.id);

  // Photo area
  var photoDiv = document.createElement('div');
  photoDiv.className = 'car-photo';

  if (car.photo) {
    var img = document.createElement('img');
    img.src = car.photo;
    img.alt = car.name;
    img.addEventListener('error', function() {
      photoDiv.innerHTML = noPhotoHTML();
      // re-append tags
      photoDiv.appendChild(makeTag(typeLabel, 'type-tag'));
      photoDiv.appendChild(makeTag(statusLabel, 'status-tag s-' + car.status));
    });
    photoDiv.appendChild(img);
  } else {
    photoDiv.innerHTML = noPhotoHTML();
  }
  photoDiv.appendChild(makeTag(typeLabel, 'type-tag'));
  photoDiv.appendChild(makeTag(statusLabel, 'status-tag s-' + car.status));

  // Specs rows
  var specsRows = [
    ['Mileage', car.mileage],
    ['Gearbox', car.gear],
    ['Engine',  car.engine],
    ['Fuel',    car.fuel],
    ['Colour',  car.colour],
    ['Drive',   car.drive]
  ].filter(function(r) { return r[1]; })
   .map(function(r) {
     return '<div class="spec"><span class="key">' + r[0] + '</span><span class="val">' + escHTML(r[1]) + '</span></div>';
   }).join('');

  var bodyHTML =
    '<div class="car-body">' +
      '<div class="car-stock">' + escHTML(car.stock) + '</div>' +
      '<div class="car-name">'  + escHTML(car.name)  + '</div>' +
      '<div class="car-price"><span class="cur">K</span>' + Number(car.price).toLocaleString() + '</div>' +
      '<div class="car-specs">' + specsRows + '</div>' +
      '<div class="car-checks">' + checks + '</div>' +

function escHTML(str) {
  return String(str)
    .replace(/&/g,'&amp;').replace(/</g,'&lt;')
    .replace(/>/g,'&gt;').replace(/"/g,'&quot;');
}

// ── RENDER ADMIN LIST ──
function renderAdminList() {
  var cars = loadCars();
  var list = el('admin-car-list');
  var msg  = el('no-cars-msg');
  list.innerHTML = '';

  if (!cars.length) { msg.style.display = 'block'; return; }
  msg.style.display = 'none';

  cars.forEach(function(car) {
    var row = document.createElement('div');
    row.className = 'admin-car-row';

    var thumbEl;
    if (car.photo) {
      thumbEl = document.createElement('img');
      thumbEl.className = 'admin-car-thumb';
      thumbEl.src = car.photo;
      thumbEl.alt = car.name;
      thumbEl.addEventListener('error', function() {
        var fb = document.createElement('div');
        fb.className = 'admin-car-thumb no-img';
        fb.textContent = '🚗';
        row.replaceChild(fb, thumbEl);
      });
    } else {
      thumbEl = document.createElement('div');
      thumbEl.className = 'admin-car-thumb no-img';
      thumbEl.textContent = '🚗';
    }

    var info = document.createElement('div');
    info.className = 'admin-car-info';
    info.innerHTML =
      '<div class="name">' + escHTML(car.name) + '</div>' +
      '<div class="meta">' + escHTML(car.stock) + ' · K' + Number(car.price).toLocaleString() +
      ' <span class="s-badge ' + car.status + '">' + car.status + '</span></div>';

    var actions = document.createElement('div');
    actions.className = 'admin-car-actions';

    var editBtn = document.createElement('button');
    editBtn.className = 'btn btn-sm btn-blue';
    editBtn.textContent = '✏️';
    editBtn.addEventListener('click', function() { editCar(car.id); });

    var delBtn = document.createElement('button');
    delBtn.className = 'btn btn-sm btn-danger';
    delBtn.textContent = '🗑️';
    delBtn.addEventListener('click', function() { deleteCar(car.id); });

    actions.appendChild(editBtn);
    actions.appendChild(delBtn);
    row.appendChild(thumbEl);
    row.appendChild(info);
    row.appendChild(actions);
    list.appendChild(row);
  });
}
